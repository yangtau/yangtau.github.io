block => {
  a => 2020
  b => "hello"
  c => {"world"}
}

assert(block.a == 2020)
assert(block["b"] == "hello")
assert(block.c{} == "world")

assert(this.block == block)
assert(block{} == nil)

block => {
  fn => {|x, y|
    a = 100
    x*a - y/a
  }
}

assert(block.fn.a == nil)
assert(block.fn{x=>2; y=>100} == 199)
